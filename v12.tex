\begin{align*}
\int \frac{f'(x)}{f(x)}dx &= ln |f(x)| + c \quad c \in \mathbb{R} \\
( ln | f(x)| )'   &= 
\begin{cases} 
(ln f(x))' , f(x)> 0 \\
(ln f(x))' , f(x)< 0 \\ 
\end{cases}\\
&=
\begin{cases}
\frac{1}{f(x)}f'(x),f(x) >0 \\
\frac{1}{-f(x)}-f'(x),f(x) < 0\\
\end{cases} \\
\end{align*}
